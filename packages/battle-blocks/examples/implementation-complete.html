Grammar Management

Program Serialization

// Load program from JSON
editor.loadProgram({
  variables: [],
  routines: [{
    name: "main",
    args: [],
    body: [
      { type: "move_toward", x: 100, y: 50 }
    ]
  }]
});

// Export program to JSON
const program = editor.getProgram();
console.log(JSON.stringify(program, null, 2));
